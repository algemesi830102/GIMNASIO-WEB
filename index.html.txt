<script>
  function calcularCalorias(event) {
    event.preventDefault();
    const peso = parseFloat(document.getElementById('peso').value);
    const altura = parseInt(document.getElementById('altura').value);
    const edad = parseInt(document.getElementById('edad').value);
    const sexo = document.getElementById('sexo').value;
    const actividad = parseFloat(document.getElementById('actividad').value);

    if (!peso || !altura || !edad || !sexo || !actividad) {
      alert('Por favor completa todos los campos correctamente.');
      return;
    }

    // Fórmula de Harris-Benedict revisada para TMB (Tasa Metabólica Basal)
    let tmb;
    if (sexo === 'masculino') {
      tmb = 88.362 + (13.397 * peso) + (4.799 * altura) - (5.677 * edad);
    } else {
      tmb = 447.593 + (9.247 * peso) + (3.098 * altura) - (4.330 * edad);
    }

    const mantenimiento = Math.round(tmb * actividad);
    const ganancia = mantenimiento + 400;
    const perdida = mantenimiento - 500;

    const resultado = `
      Calorías diarias aproximadas:<br>
      <ul>
        <li><strong>Mantenimiento:</strong> ${mantenimiento} kcal</li>
        <li><strong>Ganar músculo:</strong> ${ganancia} kcal</li>
        <li><strong>Perder grasa:</strong> ${perdida} kcal</li>
      </ul>
    `;
    document.getElementById('resultado').innerHTML = resultado;
  }
</script>
